__init__


__init__ is a special named function that acts as a constructor for classes in python


The first arguments is always self followed by the initialized variables. Much like this.etc = .. in JS

to have the order not matter for the parameters do the following

   def __init__(self, **kwargs):
      self._type = kwargs['type']
      self._name = kwargs['name']
      self._sound = kwargs['sound']

If you do this, you must declare as keys in main()

class Animal:

   def __init__(self, type, name, sound):

      self._type = type
      self._name = name
      self._sound = sound

   def type(self):
      return self._type

   def name(self):
      return self._name

   def sound(self):
      return self._sound

def print_animals(o):
   if not isinstance(o,Animal):
      raise TypeError('print_animal(): requires an Animal')
   else:
      print(f"The {o._type} named {o._name} makes a o._sound sound)

def main():
   a0 = Animal("Kitten","Mittens","meow")
   a1 = Animal("Dog","Henry","woof")
   print_animal(a0)
   print_animal(a1)
   print(Animal("Cow","Bobby","moo"))

if __name__ == '__main__': main()

Default value

to give default values to the parameters do the following

   def __init__(self, **kwargs):
      self._type = kwargs['type'] if 'type' in kwargs else 'kitten'
      self._name = kwargs['name'] if 'name' in kwargs else 'Mittens'
      self._sound = kwargs['sound'] if 'sound' in kwargs else 'meow'
